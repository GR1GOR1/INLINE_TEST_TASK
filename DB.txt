//Таблица записей -БД-
CREATE TABLE test_inline.post (
	userId int NOT null,
    id int PRIMARY KEY,
    title varchar(100) NOT null,
    body varchar(500) NOT null
    
)
//Таблица комментариев -БД-
CREATE TABLE test_inline.comm (
	postId int,
    FOREIGN KEY (postId) REFERENCES test_inline.post(id),
    id int PRIMARY KEY,
    name varchar(100) NOT null,
    email varchar(100) NOT null,
    body varchar(500) NOT null
    
)
